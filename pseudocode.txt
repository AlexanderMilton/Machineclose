outer:
	left++
	if(left == size - 1) // at end of list
		goto exit
	right = left // will be incremented in loop
	goto inner

inner:
	right++
	if(right == size) // after end of list. stop loop
		goto outer
	left_data = data[left]
	right_data = data[right]
	if(left_data < right_data) // don't swap
		goto inner
	data[right] = left_data // swap the data values
	data[left] = right_data
	goto inner // go again
